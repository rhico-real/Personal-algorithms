import java.util.*;
public class Number_game{
	public static void main(String[] args) {
		Scanner scanner=new Scanner(System.in);
		
		System.out.print("GIVE ME A NUMBER> ");
		int number=scanner.nextInt();
		
		for (int i=1;i<=number;i++) {
			for(int j=0;j<i;j++) {
				System.out.print("* ");
			}
			System.out.println("");
		}
	}
}
